CSSだけでバウンスアニメーション: linear()

transition + linear()

NEW
.transition-sample {
  scale: 1;
  transition: scale 0.3s ease-out;

  &:hover {
    scale: 1.4;
    transition-duration: 0.8s;
    transition-timing-function: linear(
      0, 
      1.4 35%, 
      0.85 55%, 
      1.1 75%, 
      0.95 90%, 
      1);
  }
}

animation + @keyframes

NEW
.animation-sample {
  animation: shrink 0.3s linear forwards;

  &:hover {
    animation-name: bounce-in-manual;
    animation-duration: 0.8s;
  }
}

@keyframes bounce-in-manual {
  0% { scale: 1; }
  35% { scale: 1.56; } /* 1 + (1.4-1) × 1.4 */
  55% { scale: 1.34; } /* 1 + (1.4-1) × 0.85 */
  75% { scale: 1.44; } /* 1 + (1.4-1) × 1.1 */
  90% { scale: 1.38; } /* 1 + (1.4-1) × 0.95 */
  100% { scale: 1.4; }
}

@keyframes shrink {
  from { scale: 1.4; }
  to { scale: 1; }
}
